###### 1.色彩Colors

- 所有示例的canvas都为height150 width150

```js
// 要给图形上色，有两个重要的属性可以做到

// 设置图形的填充颜色
fillStyle = color

// 设置图形轮廓的颜色
strokeStyle = color

// 输入的颜色值应该是符合CSS3颜色标准的有效字符串。下面的例子表示同一种颜色
ctx.fillStyle = 'orange'
ctx.fillStyle = '#FFA500'
ctx.fillStyle = 'rgb(255,165,0)'
ctx.fillStyle = 'rgba(255,165,0,1)'
```

###### 3.strokeStyle示例

```js
var ctx = document.getElementById('canvas').getContext('2d')
for(var i=0; i<6; i++){
    for(var j=0; j<6; j++){
        ctx.strokeStyle = `rgba(0, ${Math.floor(255-42.5*i)}, ${Math.floor(255-42.5*j)})`
        ctx.beginPath()
        ctx.arc(12.5+j*25, 12.5+i*25,10,0,Math.PI*2,true)
        ctx.stroke()
    }
}
```

###### 4.透明度blobalAlpha

```js
// 这个属性影响到canvas里所有图形的透明度，有效的值范围是0.0(完全透明度)到1.0(完全不透明)
globalAlpha = transparencyValue

var ctx = document.getElementById('canvas').getContext('2d')
// 画背景
ctx.fillStyle = '#FD0'
ctx.fillRect(0, 0, 75, 75)
ctx.fillStyle = '#6C0'
ctx.fillRect(75, 0, 75, 75)
ctx.fillStyle = '#09F'
ctx.fillRect(0, 75, 75, 75)
ctx.fillStyle = '#F30'
ctx.fillRect(75, 75, 75, 75)
ctx.fillStyle = '#FFF'

// 设置透明度(全局透明度)
ctx.globalAlpha = 0.2
// 画半透明圆
for(var i=0; i<7; i++){
    ctx.beginPath()
    ctx.arc(75, 75, 10+10*i, 0, Math.PI*2, true)
    ctx.fill()
}
```

###### 6.线性LineStyles

```js
// 设置线条宽度
lineWidth = value

// 设置线条末端样式
lineCap = type

// 设置线条与线条间结合处的样式
lineJoin = type

// 限制当两条线相交时交接处最大长度；所谓交接处长度(倾斜长度)是指线交接处内角顶点到外角顶点的长度。
miterLimit = value

// 返回一个包含当前虚拟样式，长度为非负偶数的数组
setLineDash()

// 设置当前虚线样式
setLineDash(segments)

// 设置虚线样式的起始偏移量
lineDashOffset = value
```

###### 8.lineCap属性例子

```js
// lineCap的值决定了线段端点显示的样子。它可以为下面的三种的其中之一:butt, round, square
// butt默认值：与辅助线齐平
// round: 端点处加上了半径为一半线宽的半圆
// square: 端点处加上了等宽且高度为一半线宽的方块

var ctx = document.getElementById('canvas').getContext('2d')
var lineCap = ['butt', 'round', 'square']

// 创建路径
ctx.strokeStyle = '#09f'
ctx.beginPath()
ctx.moveTo(10,10)
ctx.lineTo(140,10)
ctx.moveTo(10,140)
ctx.lineTo(140,140)
ctx.stroke()

// 画线条
ctx.strokeStyle = 'black'
for(var i=0; i<lineCap.length; i++){
    ctx.lineWidth = 15
    ctx.lineCap = lineCap[i]
    ctx.beginPath()
    ctx.moveTo(25+i*50, 10)
    ctx.lineTo(25+i*50, 140)
    ctx.stroke()
}
```

###### 9.lineJoin属性例子

```js
// lineJoin的属性值决定了图形中两线段连接处所显示的样子。它可以是这三种之一:round,bevel,miter.默认是miter
var ctx = document.getElementById('canvas').getContext('2d')
var lineJoin = ['round', 'bevel', 'miter']
ctx.lineWidth = 10

for(var i=0; i<lineJoin.length; i++){
    ctx.lineJoin = lineJoin[i]
    ctx.beginPath()
    ctx.moveTo(-5, 5 + i * 40)
    ctx.lineTo(35, 45 + i * 40)
    ctx.lineTo(75, 5 + i * 40)
    ctx.lineTo(115, 45 + i * 40)
    ctx.lineTo(155, 5 + i * 40)
    ctx.stroke()
}
```
