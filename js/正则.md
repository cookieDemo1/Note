###### 1.校验MAC

```js
String mac="00:E0:20:1C:7C:0C";  

var reg=/^[A-Fa-f\d]{2}:[A-Fa-f\d]{2}:[A-Fa-f\d]{2}:[A-Fa-f\d]{2}:[A-Fa-f\d]{2}:[A-Fa-f\d]{2}$/;
if(!reg.test(mac)){
    alert("mac地址不正确！");
}else{
    alert("验证成功!");
}
```

###### 2.校验IP

```js
let ip="192.168.1.255"
//正则校验IP合法性
let reg2 = /^(\d{1,2}|1\d\d|2[0-4]\d|25[0-5])\.(\d{1,2}|1\d\d|2[0-4]\d|25[0-5])\.(\d{1,2}|1\d\d|2[0-4]\d|25[0-5])\.(\d{1,2}|1\d\d|2[0-4]\d|25[0-5])$/
if(reg2.test(ip)){
  console.log("验证成功")
} else{
  console.log("IP地址格式错误！")
}

```
